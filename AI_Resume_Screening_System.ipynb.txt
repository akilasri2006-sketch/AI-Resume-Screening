    "def extract_resume_info(resume_text: str, candidate_name: str) -> dict:\n",
    "    \"\"\"\n",
    "    Uses an engineered prompt to extract structured information from a resume.\n",
    "    Returns a dictionary with skills, experience, education, and summary.\n",
    "    \"\"\"\n",
    "    prompt = f\"\"\"\n",
    "You are an expert HR analyst. Analyze the resume below and extract structured information.\n",
    "\n",
    "Return ONLY a valid JSON object with these exact keys:\n",
    "{{\n",
    "  \"name\": \"candidate full name\",\n",
    "  \"education\": \"degree and institution\",\n",
    "  \"years_experience\": number (estimate total years),\n",
    "  \"technical_skills\": [\"list\", \"of\", \"technical\", \"skills\"],\n",
    "  \"soft_skills\": [\"list\", \"of\", \"inferred\", \"soft\", \"skills\"],\n",
    "  \"domains\": [\"key domains e.g. NLP, Cloud, ML\"],\n",
    "  \"notable_achievements\": \"1-2 line summary of standout experience\"\n",
    "}}\n",
    "\n",
    "Resume:\n",
    "{resume_text}\n",
    "\n",
    "Return only the JSON. No explanation. No markdown.\n",
    "\"\"\"\n",
    "    response = client.chat.completions.create(\n",
    "        model=MODEL,\n",
    "        messages=[{\"role\": \"user\", \"content\": prompt}],\n",
    "        temperature=0.1\n",
    "    )\n",
    "    \n",
    "    raw = response.choices[0].message.content.strip()\n",
    "    # Clean any accidental markdown code fences\n",
    "    raw = re.sub(r'^```json|^```|```$', '', raw, flags=re.MULTILINE).strip()\n",
    "    \n",
    "    try:\n",
    "        return json.loads(raw)\n",
    "    except json.JSONDecodeError:\n",
    "        print(f'  ⚠️  JSON parse error for {candidate_name}. Raw response: {raw[:200]}')\n",
    "        return {}\n",
    "\n",
    "\n",
    "# ─── Run extraction for all candidates ───\n",
    "print('🔍 Extracting resume information...\\n')\n",
    "extracted_data = {}\n",
    "\n",
    "for name, resume_text in resumes.items():\n",
    "    print(f'  Processing: {name}...')\n",
    "    info = extract_resume_info(resume_text, name)\n",
    "    extracted_data[name] = info\n",
    "    print(f'  ✅ Done — Found {len(info.get(\"technical_skills\", []))} technical skills')\n",
    "\n",
    "print('\\n✅ Extraction complete for all candidates!')"
   ]

    "def score_candidate(resume_text: str, jd_text: str, candidate_name: str) -> dict:\n",
    "    \"\"\"\n",
    "    Uses an engineered prompt to score how well a candidate matches a job description.\n",
    "    Incorporates bias-aware design — evaluates skills and experience, not demographics.\n",
    "    Returns a structured score with explanation.\n",
    "    \"\"\"\n",
    "    prompt = f\"\"\"\n",
    "You are a fair, unbiased AI recruitment assistant.\n",
    "Evaluate the candidate's resume against the job description provided.\n",
    "\n",
    "IMPORTANT — Bias-Aware Instructions:\n",
    "- Evaluate ONLY on skills, experience, and qualifications\n",
    "- Do NOT factor in name, gender, age, or location\n",
    "- Focus on transferable skills, not just exact keyword matches\n",
    "\n",
    "Return ONLY a valid JSON object:\n",
    "{{\n",
    "  \"overall_score\": number from 0 to 100,\n",
    "  \"skills_match_score\": number from 0 to 100,\n",
    "  \"experience_score\": number from 0 to 100,\n",
    "  \"education_score\": number from 0 to 100,\n",
    "  \"matched_skills\": [\"skills that match the JD\"],\n",
    "  \"missing_skills\": [\"important skills from JD not found in resume\"],\n",
    "  \"strengths\": \"2-3 sentence summary of candidate strengths for this role\",\n",
    "  \"gaps\": \"1-2 sentence summary of gaps or concerns\",\n",
    "  \"recommendation\": \"Shortlist\" or \"Maybe\" or \"Reject\",\n",
    "  \"reason\": \"1 sentence reason for recommendation\"\n",
    "}}\n",
    "\n",
    "Job Description:\n",
    "{jd_text}\n",
    "\n",
    "Candidate Resume:\n",
    "{resume_text}\n",
    "\n",
    "Return only the JSON. No explanation. No markdown.\n",
    "\"\"\"\n",
    "    response = client.chat.completions.create(\n",
    "        model=MODEL,\n",
    "        messages=[{\"role\": \"user\", \"content\": prompt}],\n",
    "        temperature=0.1\n",
    "    )\n",
    "    \n",
    "    raw = response.choices[0].message.content.strip()\n",
    "    raw = re.sub(r'^```json|^```|```$', '', raw, flags=re.MULTILINE).strip()\n",
    "    \n",
    "    try:\n",
    "        return json.loads(raw)\n",
    "    except json.JSONDecodeError:\n",
    "        print(f'  ⚠️  JSON parse error for {candidate_name}')\n",
    "        return {}\n",
    "\n",
    "\n",
    "# ─── Score all candidates ───\n",
    "print('📊 Scoring candidates against job description...\\n')\n",
    "scores = {}\n",
    "\n",
    "for name, resume_text in resumes.items():\n",
    "    print(f'  Scoring: {name}...')\n",
    "    result = score_candidate(resume_text, job_description, name)\n",
    "    scores[name] = result\n",
    "    print(f'  ✅ Score: {result.get(\"overall_score\", \"N/A\")}/100 — {result.get(\"recommendation\", \"N/A\")}')\n",
    "\n",
    "print('\\n✅ Scoring complete!')"
   ]
